283  virtual void generate() override final
284  {
285  const Point grid_center = this->reconstruction->grid_properties->getCenter();
286 
287  Extrinsic extr = Extrinsic::Identity();
288 
289  size_t view_number = this->numAccepted() + this->numRejected();
290  if (this->change_random &&
291  view_number > 0 &&
292  view_number % this->n_views == 0)
293  {
294  this->axis.push_back(Direction::NullaryExpr([this](){return this->sample.uniform();}));
295  this->axis.back().normalize();
296  this->axis_to_grid_center = this->get_axis_to_grid_center();
297  }
298 
299  // Generate the position in the Axis reference frame and store the target for the axis.
300  this->call_on_generate(extr, view_number);
301  Point target_axis = Point(0.0, 0.0, extr.translation().z());
302 
303  // Rotate the position to the Grid Center's reference frame.
304  extr.translation() = this->axis_to_grid_center * extr.translation();
305 
306  // Calculate what the view's orientation should be.
307  Direction z_axis;
308  this->call_for_z_axis(z_axis, extr.translation(), target_axis);
309 
310  extr.matrix().array().block<3, 1>(0, 0) = this->axis.back();
311  extr.matrix().array().block<3, 1>(0, 1) = z_axis.cross(this->axis.back());
312  extr.matrix().array().block<3, 1>(0, 2) = z_axis;
313 
314  // Translate the position to the Grid Lower Bound's reference frame.
315  extr.translation() += grid_center;
316 
317  // For this policy, clear any views before adding the newly generated one.
318  this->views.clear();
319  this->views.push_back(extr);
320  }
