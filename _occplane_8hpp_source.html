194  void generateOccplaneViews()
195  {
196  std::vector<Eigen::Vector3d> candidate_normals, candidate_points;
197  this->binary_grid->updateOccplanes(candidate_points, candidate_normals);
198 
199  // Record the number of occplanes found. But set to zero if somehow the number of points and
200  // normals are not equal.
201  this->n_occplane = candidate_points.size() * (candidate_points.size() == candidate_normals.size());
202 
203  // Fallback to default (random) method if there are no identified occplanes.
204  if (this->n_occplane == 0)
205  {
206  this->generateDefault(true);
207  return;
208  }
209 
210  // Turn the centers & normals into the final positions for the candidate points.
211  for (size_t i = 0; i < this->n_occplane; ++i)
212  {
213  candidate_points[i] += this->radius * candidate_normals[i];
214  }
215 
216  // Labels from the Open3D DBSCAN function are [-1, 0, 1, ..., M]. We use the labels as an index into the clusters
217  // vector so we add one because of the first noise label is -1 and and then add one again to get the total size.
218  std::vector<int32_t> labels = open3d::geometry::PointCloud(candidate_points).ClusterDBSCAN(this->eps, this->min_points);
219  this->n_clusters = *(std::max_element(labels.begin(), labels.end())) + 2;
220 
221  // Fallback to default (random) method if is only one cluster and it is noise.
222  if (this->n_clusters == 1 && labels.at(0) == -1)
223  {
224  this->generateDefault(true);
225  return;
226  }
227 
228  // Create the clusters vector and index each occplane point into it, sum points and normals.
229  // The normals are inverted to the pose points from the candidate point to the occplane voxel.
230  std::vector<Cluster> clusters(this->n_clusters);
231  for (size_t i = 0; i < this->n_occplane; ++i)
232  {
233  // Labels vector is still [-1, 0, 1, ... (n_clusters - 2)]. So we add one
234  // to get the index equivelent for the cluster.
235  size_t c = labels[i] + 1;
236  clusters[c].n_points += 1;
237  clusters[c].center += candidate_points[i];
238  clusters[c].normal -= candidate_normals[i];
239  }
240 
241  // Label each cluster. Now labels start from 0 rather than -1.
242  int c = -1;
243  for (auto& vector_item : clusters)
244  {
245  vector_item.label = ++c;
246  vector_item.center /= vector_item.n_points;
247  vector_item.normal /= vector_item.n_points;
248  }
249 
250  // Sort in order of cluster size unless the cluster is labeled as noise. Once sorted, remove
251  // the last element if its label is noise.
252  auto sort_clusters = [](const Cluster& a, const Cluster& b) {
253  return (a.n_points * (a.label != 0)) > (b.n_points * (b.label != 0));
254  };
255  std::sort(clusters.begin(), clusters.end(), sort_clusters);
256  if (clusters.back().label == 0) { clusters.pop_back(); }
257  if (static_cast<int>(clusters.size()) > this->keep_top_n)
258  {
259  clusters.resize(this->keep_top_n);
260  }
261 
262  // Transform each cluster into a view and add it to the view list.
263  for (const auto& vector_item : clusters)
264  {
265  Extrinsic extr = Extrinsic::Identity();
266  extr.translation() = vector_item.center.cast<float>();
267  extr.rotate(vector_math::get_rotation_to_orient_z_axis(extr, (vector_item.center + vector_item.normal).cast<float>()));
268  this->views.push_back(extr);
269  }
270  }
