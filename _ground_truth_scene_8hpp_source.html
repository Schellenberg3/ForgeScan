197  void makeXDMF(std::filesystem::path fpath, const std::string& occupancy_dset_path = std::string(),
198  const std::string& tsdf_dset_path = std::string()) const
199  {
200  const std::string hdf5_fname = fpath.filename().string();
201  fpath.replace_extension(FS_XDMF_FILE_EXTENSION);
202 
203  const size_t num_voxels = this->grid_properties->getNumVoxels();
204 
205  Point lower = Point::Zero();
206  lower.array() -= 0.5 * this->grid_properties->resolution;
207 
208  GridSize adjsz = this->grid_properties->size.array() + 1;
209  adjsz.reverseInPlace();
210 
211  std::ofstream file;
212  file.exceptions( std::ofstream::failbit | std::ofstream::badbit );
213 
214  try
215  {
216  file.open(fpath);
217  file << std::fixed << std::setprecision(8);
218  utilities::XDMF::writeHeader(file);
219  utilities::XDMF::writeVoxelGridHeader(
220  file, this->grid_properties->resolution,
221  adjsz.x(), adjsz.y(), adjsz.z(),
222  lower.x(), lower.y(), lower.z()
223  );
224 
225  if (this->true_occupancy != nullptr && occupancy_dset_path.empty() == false)
226  {
227  utilities::XDMF::writeVoxelGridAttribute(
228  file,
229  this->true_occupancy->getTypeName(),
230  utilities::XDMF::makeDataPath(hdf5_fname, occupancy_dset_path),
231  getNumberTypeXDMF(this->true_occupancy->type_id),
232  getNumberPrecisionXDMF(this->true_occupancy->type_id),
233  num_voxels
234  );
235  }
236 
237  if (this->true_tsdf != nullptr && tsdf_dset_path.empty() == false)
238  {
239  utilities::XDMF::writeVoxelGridAttribute(
240  file,
241  this->true_tsdf->getTypeName(),
242  utilities::XDMF::makeDataPath(hdf5_fname, tsdf_dset_path),
243  getNumberTypeXDMF(this->true_tsdf->type_id),
244  getNumberPrecisionXDMF(this->true_tsdf->type_id),
245  num_voxels
246  );
247  }
248 
249  utilities::XDMF::writeVoxelGridFooter(file);
250  utilities::XDMF::writeFooter(file);
251  file.close();
252  }
253  catch (const std::ofstream::failure&)
