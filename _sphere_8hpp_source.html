126  void generateUniform(Extrinsic& dest)
127  {
128  // Golden angle in radians. See: https://en.wikipedia.org/wiki/Golden_angle
129  static const float golden_angle_radians = M_PI * (std::sqrt(5) - 1);
130 
131  // Avoids division by zero errors is n_view_requested is 1.
132  static const float nearly_one = 1 - std::numeric_limits<float>::epsilon();
133 
134  const size_t view_number = this->numAccepted() + this->numRejected();
135  if (this->n_view_requested <= view_number)
136  {
137  // One we have reached the last view we may generate we switch to random sampling.
138  this->call_on_generate = std::bind(&Sphere::generateRandom, this, std::placeholders::_1);
139  this->generateRandom(dest);
140  return;
141  }
142 
143  float y = 1 - (this->view_order[view_number] / ((float)n_view_requested - nearly_one)) * 2;
144  float r_y = std::sqrt(1 - y*y);
145 
146  float theta = golden_angle_radians * this->view_order[view_number];
147 
148  float x = std::cos(theta) * r_y;
149  float z = std::sin(theta) * r_y;
150 
151  Point position(x, y, z);
152  position *= this->radius;
153 
154  dest.translation() = position;
155  }
