11 // *********************************************************************************************************************** //
12 // * This implements the ray traversal of a VoxelGrid, as used by the Reconstruction class. * //
13 // * * //
14 // * The methods in this file follow the the Amanatides-Woo algorithm for fast voxel traversal. See: * //
15 // * http://www.cse.yorku.ca/~amana/research/grid.pdf * //
16 // * * //
17 // * Essentially, this treats the input ray as a parametrized line between the start and end point. The algorithm must * //
18 // * decide what direction - X, Y, or Z - to step in. For each direction, the track a parameter, `dist`, for the * //
19 // * distance traveled from the starting voxel in that direction. The direction that has the smallest `dist` value is * //
20 // * selected to `step` in. This either increments or decrements the index in that direction. * //
21 // * * //
22 // * When a step is take the `dist` value for that direction is updated. This update, `delta`, is equal to the * //
23 // * voxel resolution in that direction divided by the normal of the ray. This describes how many units of the vector * //
24 // * must be walked to traverse one voxel unit. * //
25 // * * //
26 // * For example if the `normal` in a direction is large (meaning the vector nearly points along the direction), the * //
27 // * `delta` will be small and the algorithm will take many steps in that direction before choosing another one. * //
28 // * * //
29 // * It is critical for this method that the grid of voxels may is considered as an axis-aligned bounding-box (AABB). * //
30 // * This means any input data must be properly transformed before reaching this algorithm. * //
31 // * * //
32 // * The algorithm also considers that the voxel "origin" is the lower-bounding corner of the voxel. If the rest of the * //
33 // * code considers the "origin" to be the center of the voxel then a reference frame shift of half the voxel resolution * //
34 // * must be applied to the ray's start point when calculating `dist`. See `correct_traversal_parameters` for more. * //
35 // *********************************************************************************************************************** //

246 inline bool get_ray_trace(const std::shared_ptr<Trace>& ray_trace,
247  const Point& sensed, const Point& origin,
248  const std::shared_ptr<const Grid::Properties>& properties,
249  const float& dist_min, const float& dist_max)
250 {
251  static constexpr std::ptrdiff_t X = 0, Y = 1, Z = 2;
252 
253  ray_trace->clear();
254 
255  // Adjusted min and max distances so we only trace the ray while it is within the Grid's bounds.
256  float dist_min_adj, dist_max_adj;
257 
258  float length;
259  Direction normal, inv_normal;
260  vector_math::get_length_normal_and_inverse_normal(sensed, origin, length, normal, inv_normal);
261  length = std::min(length, dist_max);
262 
263  const bool valid_intersection = AABB::find_zero_bounded_intersection(properties->dimensions, sensed, inv_normal,
264  dist_min, length, dist_min_adj, dist_max_adj);
265  if (valid_intersection)
266  {
267  dist_min_adj = std::max(dist_min_adj, dist_min);
268  dist_max_adj = std::min(dist_max_adj, dist_max);
269 
270  const Point sensed_adj = sensed + normal * dist_min_adj;
271  Index c_idx = properties->pointToIndex(sensed_adj);
272 
273  const std::ptrdiff_t sign[3] = {std::signbit(normal[X]), std::signbit(normal[Y]), std::signbit(normal[Z])};
274 
275  // Direction of travel (increment or decrement) along the respective axis.
276  const int step[3] = { ray_trace_helpers::get_step(X, sign),
277  ray_trace_helpers::get_step(Y, sign),
278  ray_trace_helpers::get_step(Z, sign) };
279 
280  // The amount of distance to move one voxel length along each axis based on the ray's direction.
281  const float delta[3] = { ray_trace_helpers::get_delta(X, inv_normal, properties),
282  ray_trace_helpers::get_delta(Y, inv_normal, properties),
283  ray_trace_helpers::get_delta(Z, inv_normal, properties) };
284 
285  // Cumulative distance traveled along the respective axis.
286  float dist[3] = { ray_trace_helpers::get_dist(X, sign, c_idx, sensed_adj, inv_normal, dist_min_adj, properties),
287  ray_trace_helpers::get_dist(Y, sign, c_idx, sensed_adj, inv_normal, dist_min_adj, properties),
288  ray_trace_helpers::get_dist(Z, sign, c_idx, sensed_adj, inv_normal, dist_min_adj, properties) };
289 
290  try
291  {
292  ray_trace->emplace_back(properties->at(c_idx), dist_min_adj);
293 
294  std::ptrdiff_t i = ray_trace_helpers::get_min_dist(dist);
295  while (dist[i] <= dist_max_adj)
296  {
297  c_idx[i] += step[i];
298  ray_trace->emplace_back(properties->at(c_idx), dist[i]);
299 
300  dist[i] += delta[i];
301  i = ray_trace_helpers::get_min_dist(dist);
302  }
303  }
304  catch (const VoxelOutOfRange& e)
305  {
309  // Algorithm should never go out of bounds. But catching here dose not impact performance and prevents
310  // undefined behavior and silent errors in a VoxelGrid update where all indicies are assumed to be valid.
311  throw VoxelOutOfRange("Ray tracing failed: This should not happen. Failed with: " + std::string(e.what()));
312  }
313 
314  ray_trace->set_sensed(sensed, ray_trace_helpers::get_sensed_location(dist_min_adj, dist_max_adj));
315  }
316  return valid_intersection;
317 }
