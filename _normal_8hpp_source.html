183  void generateUniform(Extrinsic& dest, const int& i)
184  {
185  // Golden angle in radians. See: https://en.wikipedia.org/wiki/Golden_angle
186  static const float golden_angle_radians = M_PI * (std::sqrt(5) - 1);
187 
188  // Avoids division by zero errors is n_view_requested is 1.
189  static const float nearly_one = 1 - std::numeric_limits<float>::epsilon();
190 
191  float y = 1 - (i / ((float)this->n_sample - nearly_one)) * 2;
192  float r_y = std::sqrt(1 - y*y);
193 
194  float theta = golden_angle_radians * i;
195 
196  float x = std::cos(theta) * r_y;
197  float z = std::sin(theta) * r_y;
198 
199  Point position(x, y, z);
200  position *= this->radius;
201 
202  dest.translation() = position;
203  }

231  float scoreNormals(const Extrinsic& extr)
232  {
233 
234  Eigen::RowVector3f axis = extr.rotation().col(2).transpose();
235  auto optical_axis = open3d::core::eigen_converter::EigenMatrixToTensor(axis);
236 
237  this->camera->resetDepth(1.0f);
238  auto results = this->o3d_scene.CastRays(Scene::getCameraRays(this->camera, extr));
239  auto normals = results["primitive_normals"];
240 
241  normals *= optical_axis[0];
242  auto pixel_scores = normals.Sum({2});
243  pixel_scores.Abs_();
244 
245  auto mask = pixel_scores.Gt(this->min_similarity).To(open3d::core::Dtype::Float32);
246  mask.Clip_(0.0f, 1.0f);
247  pixel_scores *= mask;
248 
249  // Element-wise original normals
250  // this->camera->image = open3d::core::eigen_converter::TensorToEigenMatrixXf(pixel_scores);
251  // sensor::DepthImageProcessing::imshow(this->camera, true, "pixel_scores");
252 
253 
254  return pixel_scores.Sum({0, 1}).Item<float>();

259  void normalizeNormalScores()
260  {
261  auto sort_callable = [](const std::pair<float, Extrinsic>& a, const std::pair<float, Extrinsic>& b){ return a.first < b.first; };
262  float max_score = std::max_element(this->score_and_extr.begin(), this->score_and_extr.end(), sort_callable)->first;
263 
264  auto normalize_callable = [&max_score](std::pair<float, Extrinsic>& a){ a.first /= max_score; };
265  std::for_each(this->score_and_extr.begin(), this->score_and_extr.end(), normalize_callable);

271  void findNextBestViews()
272  {
273  std::vector<int> default_idx(this->n_sample);
274  std::iota (std::begin(default_idx), std::end(default_idx), 0);
275  std::shuffle(default_idx.begin(), default_idx.end(), utilities::RandomSampler<int>(42).gen);
276 
277  this->precomputed_views.clear();
278 
279  for (size_t n = 0; n < this->n_store; ++n)
280  {
281  int best_idx = default_idx[n];
282  float best_score = 0;
283 
284  for (size_t i = 0; i < this->score_and_extr.size(); ++i)
285  {
286  if (this->score_and_extr[i].first > 0)
287  {
288  float dist_score = scoreViewDistance(score_and_extr[i].second);
289  float score = this->alpha * this->score_and_extr[i].first + (1 - this->alpha) * dist_score;
290  if (score > best_score)
291  {
292  best_score = score;
293  best_idx = i;
294  }
295  }
296  }
297 
298  this->score_and_extr.at(best_idx).first = -1;
299  this->precomputed_views.push_back({this->precomputed_views.size(), this->score_and_extr[best_idx].second});
300  }
301  }

307  float scoreViewDistance(const Extrinsic& extr)
308  {
309  float min_dist_score = 1;
310  for (const auto& view : this->precomputed_views)
311  {
312  // Since points exist on the same sphere the similarity their optical axes, oriented to the center, are
313  // analogous to the distance. E.g., views far apart will have opposite axes and a negative cosine similarity.
314  float dist = (view.second.rotation().col(2).array() * extr.rotation().col(2).array()).sum();
315 
316  // Similarity is [-1, 1], but we normalize it then invert it. Now, most similar vectors have a score of 0
317  // and dissimilar ones a score of 1.
318  min_dist_score = std::min(min_dist_score, 1 - ((dist + 1) / 2));
319  }
320  return min_dist_score;
321  }
