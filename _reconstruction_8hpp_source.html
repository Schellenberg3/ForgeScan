236  void updateMinAndMaxDist() {
237  this->min_dist_min = 0;
238  this->max_dist_max = 0;
239 
240  for (const auto& item : this->channels)
241  {
242  // Record the most negative distance as min and most positive distance as max.
243  this->min_dist_min = std::min(this->min_dist_min, item.second->dist_min);
244  this->max_dist_max = std::max(this->max_dist_max, item.second->dist_max);
245  }
246 
247  // Verify that we never have a case where the min is greater than the max.
248  // This should never happen but it would be a critical error if it did.
249  assert(this->max_dist_max >= this->min_dist_min &&
250  "Reconstruction maximum dist max was less than the minimum dist min. "
251  "This should not be possible.");
252  }
