208  open3d::core::Tensor getVoxelCenters(const std::shared_ptr<const Grid::Properties>& grid_properties,
209  const Extrinsic& lower_bound) const
210  {
211  const int64_t nx = static_cast<int64_t>(grid_properties->size.x()),
212  ny = static_cast<int64_t>(grid_properties->size.y()),
213  nz = static_cast<int64_t>(grid_properties->size.z());
214 
215  open3d::core::Tensor voxel_centers({nx, ny, nz, 3}, open3d::core::Float32);
216 
217  Point center_lower_bound_f = Point::Zero();
218  for (int64_t z = 0; z < nz; ++z)
219  {
220  for (int64_t y = 0; y < ny; ++y)
221  {
222  for (int64_t x = 0; x < nx; ++x)
223  {
224  Point center = lower_bound * center_lower_bound_f.homogeneous();
225  voxel_centers[x][y][z][0] = center.x();
226  voxel_centers[x][y][z][1] = center.y();
227  voxel_centers[x][y][z][2] = center.z();
228  center_lower_bound_f.x() += grid_properties->resolution;
229  }
230  center_lower_bound_f.x() = 0;
231  center_lower_bound_f.y() += grid_properties->resolution;
232  }
233  center_lower_bound_f.y() = 0;
234  center_lower_bound_f.z() += grid_properties->resolution;
235  }
236  return voxel_centers;
